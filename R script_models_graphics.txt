datos_largo <- read_csv("BASE_DATOS_LARGO_2.csv",
                        locale = locale(encoding = "Latin1"))

datos_largo_z <- datos_largo %>%
  mutate(
    hora_z = scale(hora, center = TRUE, scale = TRUE),
    Luz_z  = scale(Luz, center = TRUE, scale = TRUE),
    Temp_z = scale(Temp, center = TRUE, scale = TRUE),
    NDVI_z = scale(NDVI, center = TRUE, scale = TRUE)
  )

modelo_BI <- lmer(
  BI ~ (Luz_z + Temp_z + NDVI_z) * sitio_type * temporada +
    (1 | sitio_id),
  data = datos_largo_z
)

summary(modelo_BI)

modelo_ACI <- lmer(
  ACI ~ (Luz_z + Temp_z + NDVI_z) * sitio_type * temporada +
    (1 | sitio_id),
  data = datos_largo_z
)

summary(modelo_ACI)

modelo_cantos <- glmmTMB(
  cantos ~ (Luz_z + Temp_z + NDVI_z) * sitio_type * temporada +
    (1 | sitio_id),
  family = nbinom2,
  data = datos_largo_z
)

modelo_cantos_simple <- glmer.nb(
  cantos ~ temporada * sitio_type +

  control = glmerControl(optimizer = "bobyqa")

#TABLE ANOVA
anova_BI <- as.data.frame(Anova(modelo_BI, type = 3))
anova_BI$Efecto <- rownames(anova_BI)
anova_BI$Modelo <- "BI"

anova_ACI <- as.data.frame(Anova(modelo_ACI, type = 3))
anova_ACI$Efecto <- rownames(anova_ACI)
anova_ACI$Modelo <- "ACI"

anova_cantos <- as.data.frame(Anova(modelo_cantos_simple, type = 3))
anova_cantos$Efecto <- rownames(anova_cantos)
anova_cantos$Modelo <- "Cantos"

tabla_anova <- bind_rows(anova_BI, anova_ACI, anova_cantos)

efectos_clave <- c(
  "temporada",
  "sitio_type",
  "Luz_z",
  "Temp_z",
  "NDVI_z",
  "temporada:sitio_type"
)

tabla_anova_filtrada <- tabla_anova %>%
  filter(
    (`Pr(>Chisq)` < 0.05) |
      (Efecto %in% efectos_clave)
  )


tabla_anova_final <- tabla_anova_filtrada %>%
  select(Modelo, Efecto, Chisq, Df, `Pr(>Chisq)`) %>%
  rename(
    Chi2 = Chisq,
    gl = Df,
    p_value = `Pr(>Chisq)`
  )

tabla_anova_final <- tabla_anova_final %>%
  mutate(
    p_value = ifelse(p_value < 0.001, "<0.001", round(p_value, 4)),
    Chi2 = round(Chi2, 2)
  )


tabla_anova_final <- tabla_anova_final %>%
  arrange(Modelo, p_value)


#TABLE COEFINCENT

library(broom.mixed)
library(dplyr)

coef_BI <- broom.mixed::tidy(modelo_BI, effects = "fixed") %>%
  mutate(
    Modelo = "LMM",
    Respuesta = "BI"
  )

coef_ACI <- broom.mixed::tidy(modelo_ACI, effects = "fixed") %>%
  mutate(
    Modelo = "LMM",
    Respuesta = "ACI"
  )

coef_cantos <- broom.mixed::tidy(modelo_cantos_simple, effects = "fixed") %>%
  mutate(
    Modelo = "GLMM (NB)",
    Respuesta = "Cantos"
  )

tabla_coeficientes <- bind_rows(coef_BI, coef_ACI, coef_cantos) %>%
  select(
    Modelo,
    Respuesta,
    term,
    estimate,
    std.error,
    statistic,
    p.value
  ) %>%
  rename(
    Efecto = term,
    Estimate = estimate,
    SE = std.error,
    Estadistico = statistic,
    p_value = p.value
  )

tabla_coeficientes

tabla_filtrada <- tabla_coeficientes %>%
  dplyr::filter(
    p_value < 0.05 | 
      !grepl(":", Efecto)   # mantiene efectos principales
  )

tabla_BI <- tabla_filtrada %>% filter(Respuesta == "BI")
tabla_ACI <- tabla_filtrada %>% filter(Respuesta == "ACI")
tabla_cantos <- tabla_filtrada %>% filter(Respuesta == "Cantos")

tabla_BI <- tabla_BI %>%
  arrange(p_value)

tabla_ACI <- tabla_ACI %>%
  arrange(p_value)

tabla_cantos <- tabla_cantos %>%
  arrange(p_value)
